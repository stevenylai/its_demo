package pubsub.edl.ast;
import java.util.*;

public class TEMP extends GenericTreeNode {
	public TEMP() {
		createChildren();
	}
	public String getName () {
		return "Nonterminal: TEMP";
	}
	public void adjust () {
		int i=0;
		while (i<children.size()) {
			Object child = children.get(i);
			if (child!=null && child instanceof TEMP) {
				parent.addChild((NodeOperation)children.remove(i));
			}
			else
				i++;
		}
		super.adjust();
	}
}
